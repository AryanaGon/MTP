#include <stdio.h>
#include <stdlib.h>
#include <math.h>

int soma(double *A, double *B, int n, int i)
{
	if (i<=n)
	{
		A[i]= A[i]+B[i];
		soma(A,B,n,i+1);
	}
	return (*A);
}
int raiz(double *A, int n, int i)
{
	if (i<=n)
	{
		A[i] = sqrt(A[i]);
		raiz(A, n, i+1);
	}
	return (*A);
}
int mult(double *B, int n, int i)
{
	if (i<=n)
	{
		B[i]= 3*B[i];
		mult(B,n,i+1);
	}
	return (*B);
}
int pow(double *B, int n, int i)
{
	if (i<=n)
	{
		B[i]= pow(B[i],2);
		mult(B,n,i+1);
	}
	return (*B);
}
int div(double *A,double *B, int n, int i)
{
	if (i<=n)
	{
		A[i]= B[i]/A[i];
		div(A,B,n,i+1);
	}
	return (*A);
}
